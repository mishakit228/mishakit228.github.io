Разработка многопоточных приложений
  1. Параллельная Обработка Задач:
  • Описание: Создайте приложение, которое выполняет задачи параллельно с использованием многопоточности. Например, рассмотрим ситуацию, где несколько файлов обрабатываются одновременно в разных потоках.
#include <System.Classes.hpp>
#include <Vcl.StdCtrls.hpp>
#include <System.SysUtils.hpp>
class TThreadedFileProcessor : public TThread
{
private:
AnsiString FileName;
TMemo* LogMemo;
protected:
void __fastcall Execute()
{
// Пример обработки файла в отдельном потоке
try
{
LogMemo->Lines->Add("Обработка файла: " + FileName);
// ... код обработки файла ...
LogMemo->Lines->Add("Готово: " + FileName);
}
catch (Exception &e)
{
LogMemo->Lines->Add("Ошибка в файле " + FileName + ": " + e.Message);
}
}
public:
__fastcall TThreadedFileProcessor(const AnsiString AFileName, TMemo* AMemo)
: TThread(true), FileName(AFileName), LogMemo(AMemo) {}
// Дополнительные методы, если необходимо
};
// Пример кода для создания и запуска потока
void __fastcall TMainForm::ProcessFilesButtonClick(TObject *Sender)
{
TStringList* FilesToProcess = new TStringList;
FilesToProcess->Add("file1.txt");
FilesToProcess->Add("file2.txt");
// ... добавление других файлов ...
for (int i = 0; i < FilesToProcess->Count; ++i)
{
TThreadedFileProcessor* Thread = new TThreadedFileProcessor(FilesToProcess->Strings[i], LogMemo);
Thread->FreeOnTerminate = true;
Thread->Start();
}
delete FilesToProcess;
}
